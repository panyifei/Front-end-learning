### 打印出 1 - 10000 之间的所有对称数

```javascript
// 我觉得我的性能好
function print() {
  let i = 1;
  const res = [];
  while (i <= 9) {
    res.push(i);
    res.push(i + "" + i);
    i++;
  }
  i = 1;
  let j = 0;
  while (i <= 9) {
    while (j <= 9) {
      res.push(i + "" + j + "" + i);
      res.push(i + "" + j + "" + j + "" + i);
      j++;
    }
    i++;
    j = 0;
  }
  return res;
}

new Array(10000)
  .fill()
  .map((_, i) => i + 1)
  .filter(
    a =>
      `${a}`
        .split("")
        .reverse()
        .join("") == a
  );
```
