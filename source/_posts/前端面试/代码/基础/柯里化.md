```javascript
// 简单的递归一下，柯里化的目的也就是函数式编程
function currying(fn, ...args) {
  if (args.length >= fn.length) {
    return fn(...args);
  }
  return function(...args2) {
    return currying(fn, ...args, ...args2);
  };
}

// 做一个加法
function add(...args) {
  let sum = 0;
  const innerAdd = (...args) => {
    args.forEach(i => (sum += i));
    return innerAdd;
  };
  innerAdd.toString = () => sum;
  return innerAdd(...args);
}
```
