### 实现一个冒泡排序

```javascript
// 优化方法就是如果没有变，就不执行了
function bubble(nums) {
  let j = nums.length - 1;
  helper(0);

  function helper(i) {
    let has = false;
    for (let j = nums.length - 1; j > i; j--) {
      if (nums[j] < nums[j - 1]) {
        const temp = nums[j - 1];
        nums[j - 1] = nums[j];
        nums[j] = temp;
        has = true;
      }
    }
    if (j > i && has) {
      helper(i + 1);
    }
  }
}
```
